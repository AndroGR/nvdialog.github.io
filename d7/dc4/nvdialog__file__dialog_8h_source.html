27/**
28 * @brief An opaque file dialog type, representing either a file
29 * dialog that can be used to open a file or save a file.
30 * @since v0.1.0
31 */
32typedef struct _NvdFileDialog NvdFileDialog;
33
34/**
35 * @brief Creates a new, empty @ref NvdFileDialog to be used for
36 * retrieving a file and returns it.
37 *
38 * This creates a new file dialog and returns it. In order to use it, you must call
39 * @ref nvd_get_file_location and pass a pointer to write the file location to.
40 * @note This function creates problems with the Adwaita backend. Those problems should be
41 * fixed by the time v0.6.0 is officially released.
42 *
43 * @param title A string to put as the dialog title.
44 * @param file_extensions Unused legacy parameter, just pass NULL in here.
45 * @returns An empty @NvdFileDialog object if successful, otherwise NULL and an error retrievable
46 * through @ref nvd_get_error is set.
47 */
48NVD_API NvdFileDialog *nvd_open_file_dialog_new(const char *title,
49 const char *file_extensions);
50
51/**
52 * @brief Creates a new, empty @ref NvdFileDialog object that will be used to save a
53 * file in the selected (from the user) location.
54 * @param title The title of the dialog, default is "Save file"
55 * @param default_filename The default filename to use for saving.
56 * @return An @ref NvdFileDialog on success, otherwise NULL and @ref nvd_get_error should
57 * be called to get the failure reason.
58 */
59NVD_API NvdFileDialog *nvd_save_file_dialog_new(const char *title,
60 const char *default_filename);
61/**
62 * @brief Returns the filesystem path chosen through the @ref NvdFileDialog passed.
63 *
64 * This function will return the path on the filesystem from the dialog chosen, that you can
65 * then use to either open or save the file given. It works with both save and open file dialog
66 * types.
67 *
68 * @sa nvd_open_file_dialog_new
69 * @param dialog The file dialog to take the filename from.
70 * @param savebuf A pointer to a buffer where the path will be written, must not be NULL.
71 */
72NVD_API void nvd_get_file_location(NvdFileDialog* dialog, const char** savebuf);
73
74/**
75 * @brief Returns the raw object behind the dialog.
76 * @param dialog The dialog to retrieve the object from.
77 * @return void* The raw toolkit-created object.
78 */
79NVD_API void *nvd_open_file_dialog_get_raw(NvdFileDialog *dialog);
